课时补充

转义字符 : \

var str = 'abcd\'efg'
var str = 'abcd\'\nefg':换行
正常情况：一个回车：\r\n
\t一个tab建
\r行间数
\i忽视大小写
\g全局匹配
var reg = /abc/i
var str = 'aBcd'
console.log(reg.test(str)) //true
\i忽视大小写
\g全局匹配
\m多行匹配
var reg = /ab/g
var str = 'abaabababababababab'
console.log(str.match(reg))
var reg = /^a/g;
var str = 'abcda'
console.log(str.match(reg)) //返回值只有一个a
var reg = /^a/gm;//多行匹配
var str = 'abcfdkslfjd\nafjdlkjfsldkfj'
console.log(str.match(reg))//返回值为2个a
reg.test(str) 只能判断str里面是否含有正则表达式里面的规则:返回值是true或false
str.match(reg) 可以把str里面符合正则表达式的规则里面的字符返回出来
表达式：[],一个[]是一位,[]里面的是区间
var reg = /[1234567890][1234567890][134567890]/g
var str = '123fsjd4654lk564564j'
console.log(str.match(reg)) //返回值：["123", "465", "564", "564"]
var reg = /[ab][cd][d]/g;
var str = 'abcd'
console.log(str.match(reg))//bcd匹配
var reg = /[0-9A-z][cd][d]/g
var str = 'abdsfdsfdsfdsfcd'
console.log(str.match(reg))
var reg = /[^a][^b][^c]/g//[^]代表的是非
var str = 'abcddskfjlksd'
console.log(str.match(reg))

var reg = /(abc|bcd)/g //匹配abc或bcd
var str = 'abcbcd'
console.log(str.match(reg))
元字符与表达式类似 \w:单词字符
\w === [0-9A-z]
\W === [^\w]
var reg = /\wcd2/g
var str = 'bcd21'
console.log(str.match(reg))
可以在表达式里面写元字符
\w [0-9A-z]
\d数字 [0-9]
\s [\t\n\r\v\f] 空白字符
\b 单词边界
. === [^\r\n]
var str = 'abc def ghi'
var reg = /c\b/g
console.log(str.match(reg)) //true
